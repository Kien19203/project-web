CREATE DATABASE handmadestore;
USE handmadestore;
-- Tạo bảng role
CREATE TABLE role (
    id VARCHAR(255) PRIMARY KEY,
    name VARCHAR(255)
);

-- Tạo bảng user
CREATE TABLE user (
    id VARCHAR(255) PRIMARY KEY,
    name NVARCHAR(255),
    phoneNumber VARCHAR(255),
    email VARCHAR(255),
    password VARCHAR(255),
    createDate DATETIME,
    status VARCHAR(255),
    roleID VARCHAR(255),
    FOREIGN KEY (roleID) REFERENCES role(id)
);
